<script>
  const form = document.getElementById('filter-form');
        const table = document.getElementById('item-table');

        form.addEventListener('submit', (event) => {
            event.preventDefault(); 

            const name = document.getElementById('item-name-input').value.toLowerCase();
            const listings = document.getElementById('listings-input').value.toLowerCase();
            const sellPrice = document.getElementById('sell-price-input').value.toLowerCase();
            const buyorderPrice = document.getElementById('buyorder-price-input').value.toLowerCase();
            const type = document.getElementById('type-input').value.toLowerCase();

            for (let i = 0; i < table.tBodies[0].rows.length; i++) {
                const row = table.tBodies[0].rows[i];
                const rowData = row.textContent.toLowerCase();

                if (rowData.includes(name) &&
                    rowData.includes(listings) &&
                    rowData.includes(sellPrice) &&
                    rowData.includes(buyorderPrice) &&
                    rowData.includes(type)) {
                    row.style.display = '';
                } else {
                    row.style.display = 'none';
                }
            }
        });
<\script>
